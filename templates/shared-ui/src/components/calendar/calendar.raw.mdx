# Calendar

Calendar component for date selection with various modes and styles.

## Example

```tsx
import * as React from "react"
import { Calendar } from "@repo/{{ package_name | kebab_case }}"

export function CalendarDemo() {
  const [date, setDate] = React.useState<Date | undefined>(new Date())

  return (
    <Calendar
      mode="single"
      selected={date}
      onSelect={setDate}
      className="rounded-md border shadow"
    />
  )
}
```
