# Card

Displays a card with header, content, and footer.

## Example

```tsx
import { Card, CardHeader, CardFooter, CardTitle, CardDescription, CardContent } from '@repo/shared-ui/components'

// Basic usage
<Card>
  <CardHeader>
    <CardTitle>Card Title</CardTitle>
    <CardDescription>Card Description</CardDescription>
  </CardHeader>
  <CardContent>
    Card Content
  </CardContent>
  <CardFooter>
    Card Footer
  </CardFooter>
</Card>
```
