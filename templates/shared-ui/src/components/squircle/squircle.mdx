# Squircle

A container component that wraps content in a squircle shape - a squared circle with smooth corners.

## Example

```tsx
import { SquircleIconContainer } from '@repo/shared-ui/components'
import * as Lucide from 'lucide-react'

// Small squircle
<SquircleIconContainer size="small" className="bg-blue-500/10 text-blue-500">
  <Lucide.Star className="size-4 relative z-10" />
</SquircleIconContainer>

// Large squircle
<SquircleIconContainer size="large" className="bg-green-500/10 text-green-500">
  <Lucide.Heart className="size-5 relative z-10" />
</SquircleIconContainer>
```
