# Chart

A responsive chart component built with Recharts that supports various chart types and themes.

## Example

```tsx
import { ChartContainer } from '#/components/base-ui'
import { LineChart, Line } from 'recharts'

const data = [
  { name: 'Jan', total: 167 },
  { name: 'Feb', total: 245 },
]

const config = {
  total: {
    theme: {
      light: 'hsl(var(--primary))',
      dark: 'hsl(var(--primary))',
    },
  },
}

<ChartContainer config={config}>
  <LineChart data={data}>
    <Line
      type="monotone"
      dataKey="total"
      strokeWidth={2}
      style={{
        stroke: 'var(--color-total)',
      }}
    />
  </LineChart>
</ChartContainer>
```
