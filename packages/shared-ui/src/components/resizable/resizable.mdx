# Resizable

A component that allows users to resize panels by dragging a handle.

## Example

```tsx
import { ResizablePanelGroup, ResizablePanel, ResizableHandle } from '@repo/shared-ui/components'

<ResizablePanelGroup direction="horizontal">
  <ResizablePanel defaultSize={25}>
    <div>Sidebar</div>
  </ResizablePanel>
  <ResizableHandle withHandle />
  <ResizablePanel defaultSize={75}>
    <div>Content</div>
  </ResizablePanel>
</ResizablePanelGroup>
```
