# Drawer

A panel that slides out from the edge of the screen.

## Example

```tsx
import {
  Drawer,
  DrawerContent,
  DrawerDescription,
  DrawerFooter,
  DrawerHeader,
  DrawerTitle,
  DrawerTrigger,
} from '@repo/shared-ui'

<Drawer>
  <DrawerTrigger>Open</DrawerTrigger>
  <DrawerContent>
    <DrawerHeader>
      <DrawerTitle>Edit Profile</DrawerTitle>
      <DrawerDescription>
        Make changes to your profile here.
      </DrawerDescription>
    </DrawerHeader>
    <div>Content</div>
    <DrawerFooter>
      <Button>Save changes</Button>
    </DrawerFooter>
  </DrawerContent>
</Drawer>
```
